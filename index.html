   if(t.inf>0){

   }

   if(t.s>=1){t.cone-=dt;

    if(t.cone<=0){t.cone=rnd(CONE_EVERY*0.7,CONE_EVERY*1.3);

     if(cones.length<32){

      var snagNear=false;

      var nthrow=2+(snagNear?1:0);

      for(var cc=0;cc<nthrow;cc++)castCone(t);

      sfxCone();}}}

  }else if(t.state==='burning'){burning++;

   if(t.retard>0)t.retard=Math.max(0,t.retard-dt);

   t.bt+=dt*(t.adapt?0.55:1)*(t.retard>0?0.45:1)*(rainDur>0?0.5:1);

   if(rainDur>0&&Math.random()<dt*0.4){t.state='alive';t.bt=0;burst(t.x,t.gy-t.h*t.s*0.6,7,'drop');continue;}

   if(Math.random()<dt*8)parts.push(P(t.x+rnd(-9,9)*t.s,t.gy-t.h*t.s*rnd(0.35,0.95),'flame'));

   if(Math.random()<dt*5)parts.push(P(t.x+rnd(-6,6),t.gy-t.h*t.s,'smoke'));

   for(var j=0;j<trees.length;j++){var o=trees[j];if(o.state!=='alive')continue;

    var dx=o.x-t.x;var d=Math.abs(dx);var R=92;

    if(d>0&&d<R){

     if(lineBlocks(t.x,o.x))continue;

     var dir=(dx*wind>0)?(1+Math.abs(wind)*1.6):(1-Math.abs(wind)*0.5);

     var pr=SPREAD*dir*(1-d/R)*dt*(1+elapsed/170);

     if(Math.random()<pr)ignite(o);}}

  }else{

   if(t.sm<6&&Math.random()<dt*0.8){t.sm++;parts.push(P(t.x,t.gy-t.h*0.4,'smoke'));}

  }

 }

 acc+=live*dt*0.5;

 if(acc>=1){var g=Math.floor(acc);score+=g;acc-=g;}

 if(live+burning===0){gameOver();return;}

 if(rainDur<=0){

  spawnT-=dt;

  var hit=false;

    if(rainDur>0&&Math.random()<0.6)fire=false;

    if(fire)ignite(tr);

    hit=true;break;}}

  if(hit){embers.splice(k,1);continue;}

  if(e.y>groundY(e.x)-3){burst(e.x,e.y,2,'smoke');embers.splice(k,1);continue;}

  if(e.x<-44||e.x>W+44)embers.splice(k,1);

 }

 beetleCheck-=dt;

 if(beetleCheck<=0){beetleCheck=2.5;

  if(snagCount()>=2&&beetles.length<10&&beetlePend<=0&&outbreakWarn<=0){

   outbreakWarn=1.6;beetlePend=2+Math.min(4,Math.floor(snagCount()/2));beetlePendT=1.0;sfxBeetle();}}

 if(outbreakWarn>0)outbreakWarn-=dt;

 if(beetlePend>0){beetlePendT-=dt;if(beetlePendT<=0&&outbreakWarn<=0.7){beetlePendT=0.25;beetlePend--;spawnBeetle();}}

 wpCheck-=dt;

  for(var wi=0;wi<trees.length;wi++){if(trees[wi].state==='alive'&&trees[wi].inf>0.5){anyInf=true;break;}}

  if((pressure>=2||(anyInf&&snagCount()>=2))&&woodpeckers.length<2){

 else{

  else{rainT-=dt;if(rainT<=0){rainWarn=2.2;rainT=rnd(28,44);}}
